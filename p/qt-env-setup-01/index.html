cmake_minimum_required(VERSION 3.10)  #cmakeзүҲжң¬жңҖдҪҺ3.10

# и®ҫзҪ®CMAKEе·Ҙе…·й“ҫ
set(CMAKE_TOOLCHAIN_FILE "D:/cppsoft/vcpkg/scripts/buildsystems/vcpkg.cmake")

# иҺ·еҸ–еҪ“еүҚеӯҗCMakeж–Үд»¶жүҖеңЁзӣ®еҪ•зҡ„е®Ңж•ҙи·Ҝеҫ„ 
get_filename_component(CURRENT_DIR_PATH "${CMAKE_CURRENT_LIST_DIR}" ABSOLUTE) 
# д»Һе®Ңж•ҙи·Ҝеҫ„дёӯжҸҗеҸ–зӣ®еҪ•еҗҚз§° 
get_filename_component(CURRENT_DIR_NAME "${CURRENT_DIR_PATH}" NAME)

# и®ҫзҪ®йЎ№зӣ®еҗҚз§°
project(${CURRENT_DIR_NAME})

set(CMAKE_CXX_STANDARD 11) 

if(MSVC)
	add_compile_options(/Zc:__cplusplus)
endif()

# иҮӘеҠЁеҢ…еҗ«еҪ“еүҚзӣ®еҪ•
SET(CMAKE_INCLUDE_CURRENT_DIR ON)

# иҮӘеҠЁз”ҹжҲҗmocж–Үд»¶
SET(CMAKE_AUTOMOC ON)
# иҮӘеҠЁз”ҹжҲҗrccж–Үд»¶
SET(CMAKE_AUTORCC ON)
# иҮӘеҠЁз”ҹжҲҗuicж–Үд»¶
SET(CMAKE_AUTOUIC ON)

# жҹҘжүҫQt5еә“
find_package(Qt5 COMPONENTS Core Widgets Gui REQUIRED)

file(GLOB UI_FILES "${CMAKE_CURRENT_SOURCE_DIR}/*.ui")
file(GLOB RCC_FILES "${CMAKE_CURRENT_SOURCE_DIR}/*qrc")
file(GLOB HEADER_FILES "${CMAKE_CURRENT_SOURCE_DIR}/*.h" "${CMAKE_CURRENT_BINARY_DIR}/*.h")
file(GLOB CPP_FILES "${CMAKE_CURRENT_SOURCE_DIR}/*.cpp")

qt5_wrap_ui(UI_HEADERS ${UI_FILES})

source_group("Form Files" FILES ${UI_FILES})
source_group("Resource Files" FILES ${RCC_FILES})


# win32иЎЁзӨәе…ій—ӯжҺ§еҲ¶еҸ°зӘ—еҸЈ
add_executable(${PROJECT_NAME} WIN32 ${HEADER_FILES} ${CPP_FILES} ${UI_FILES} ${RCC_FILES})

# й“ҫжҺҘеә“
target_link_libraries(${PROJECT_NAME} 
	Qt5::Widgets
	Qt5::Core
	Qt5::Gui
)

й…ҚзҪ®Qt VS Tools

еҶҚvs2019жү©еұ•е•Ҷеә—йҮҢжҗңзҙўQtпјҢдёӢиҪҪдёӢйқўиҝҷдёӘжҸ’д»¶пјҲеҰӮжһңжІЎжңүйӯ”жі•пјҢиҝһдёҚдёҠжү©еұ•е•Ҷеә—пјҢжүҫе…¶д»–ж–№ејҸдёӢиҪҪпјү зӮ№еҮ»Qt VS Toolsзҡ„options е°ҶиҝҷдәӣйҖүйЎ№и®ҫдёәtrueпјҲеҰӮжһңдҪ д№ӢеҗҺдёҚжғідҪҝз”ЁQt Designerд№ҹеҸҜд»ҘдёҚдҪҝз”Ёпјү

жӯӨж—¶дҫҝеҸҜд»ҘдҪҝз”ЁQt Designerжү“ејҖ .ui ж–Үд»¶ пјҲеҰӮжһңжү“ејҖеӨұиҙҘпјҢеҺҹеӣ дј°и®ЎжҳҜжүҫдёҚеҲ°Qt DesingerпјҢеҸій”® –> жү“ејҖж–№ејҸпјҢиҮӘе·ұй…ҚзҪ®дёҖдёӘQt DesingerпјҢиҝҷйҮҢжҳҜQt DesingerйңҖиҰҒиҮӘе·ұеҺ»е®ҳзҪ‘жҲ–е…¶д»–ең°ж–№дёӢиҪҪпјү

qt жәҗз Ғи°ғиҜ•й…ҚзҪ®

еңЁејҖеҸ‘иҝҮзЁӢдёӯпјҢеҰӮжһңжғійҖҡиҝҮж–ӯзӮ№иҝӣе…ҘеҲ° qt жәҗз ҒдёӯпјҢжҲ‘д»¬еҸҜд»Ҙй…ҚзҪ®pdbж–Үд»¶гҖӮ

PDBж–Үд»¶пјҲProgram Database FileпјүжҳҜMicrosoft Visual Studioзј–иҜ‘з”ҹжҲҗзҡ„и°ғиҜ•дҝЎжҒҜж–Үд»¶пјҢдё»иҰҒз”ЁдәҺеӯҳеӮЁзЁӢеәҸзҡ„з¬ҰеҸ·дҝЎжҒҜпјҲеҰӮеҮҪж•°еҗҚгҖҒеҸҳйҮҸеҗҚгҖҒиЎҢеҸ·зӯүпјүпјҢдҪҝи°ғиҜ•еҷЁиғҪе°ҶжңәеҷЁд»Јз ҒдёҺжәҗд»Јз ҒеҜ№еә”пјҢд»ҺиҖҢиҫ…еҠ©ејҖеҸ‘дәәе‘ҳе®ҡдҪҚй—®йўҳгҖӮ

еҸій”®зӮ№еҮ»vs 2019 и§ЈеҶіж–№жЎҲ иҝӣе…ҘеұһжҖ§дёӯ,ж·»еҠ зӣёеә”зҡ„жәҗз Ғи·Ҝеҫ„

з„¶еҗҺи®ҫзҪ®pdbж–Үд»¶пјҢpdbж–Үд»¶йңҖиҰҒдҪҝз”ЁDebugзј–иҜ‘з”ҹжҲҗгҖӮз”ұдәҺжҲ‘д»¬дҪҝз”Ёvcpkgе®үиЈ…дәҶqt5пјҢе®ғе·Із»Ҹз”ҹжҲҗдәҶpdbж–Үд»¶

и·Ҝеҫ„пјҡD:\cppsoft\vcpkg\installed\x64-windows\debug\bin зӮ№еҮ»е·Ҙе…· –> йҖүйЎ№ –> и°ғиҜ• —> з¬ҰеҸ·
